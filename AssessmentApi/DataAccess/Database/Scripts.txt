### Employee Table

// Create
CREATE TABLE IF NOT EXISTS Employee 
(id INTEGER PRIMARY KEY AUTOINCREMENT, 
name VARCHAR(50), 
phoneNumber VARCHAR(12), 
email VARCHAR(75));

// Select all 
SELECT * FROM Employee;

// Add Column fax
ALTER TABLE Employee ADD COLUMN fax VARCHAR(12);

// Add Column isActive
ALTER TABLE Employee ADD COLUMN isActive INTEGER DEFAULT 1;

// Add Column jobTitle
ALTER TABLE Employee ADD COLUMN jobTitle VARCHAR(50) DEFAULT 'Engineer';

### Project Table

// Create
CREATE TABLE IF NOT EXISTS Project
(id INTEGER PRIMARY KEY AUTOINCREMENT, 
name VARCHAR(50),
contact VARCHAR(50),
date VARCHAR(10));

// Select All
SELECT * FROM Project;

// Add Column status
ALTER TABLE Project ADD COLUMN status INTEGER DEFAULT 0;

// Add Column description
ALTER TABLE Project ADD COLUMN description VARCHAR(100);

// Add Column isDeleted
ALTER TABLE Project ADD COLUMN isDeleted INTEGER DEFAULT 0;
